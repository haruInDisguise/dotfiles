 77 -- KEYMAPS
 78 local global_set_keymap = vim.api.nvim_set_keymap
 79 local mapping_options   = { silent = true, noremap = true}
 80 
 81 -- Disable search result highlightinh
 82 global_set_keymap('n', '<leader><esc>', '<cmd>nohlsearch<CR>', mapping_options)
 83 
 84 -- Save file
 85 global_set_keymap('n', '<leader>w', '<cmd>w<CR>', mapping_options)
 86 
 87 -- Quit buffer or exit
 88 global_set_keymap('n', '<leader>q', '<cmd>bd<CR>', mapping_options)
 89 
 90 -- Remap H and L to jump to end/beginning of line respectivley
 91 global_set_keymap('n', 'H', '$', mapping_options)
 92 global_set_keymap('n', 'L', '0', mapping_options)
 93 
 94 -- Centering search results
 95 global_set_keymap('n', 'n', 'nzz', mapping_options)
 96 global_set_keymap('n', 'N', 'Nzz', mapping_options)
 97 
 98 -- Switch to last buffer and respecitve cursor position
 99 global_set_keymap('n', '<bs>', '<c-^>`”zz', mapping_options)
100 
101 -- Save the need to press shift when using US keymap...
102 global_set_keymap('n', ';', ':', mapping_options)
